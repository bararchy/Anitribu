//
//  Auto-Leveling Templates:
//
//	These attempt to buy each Trait (Attribute/Ability/Discipline), in order.  For example, if you have:
//		Level { "Strength"	"3" }
//		Level { "Brawl"		"2" }
//		Level { "Strength"	"4" }
//	The system will attempt to buy Strength up to 3, then Brawl up to 2, and then Strength to 4.  If it
//	cannot pay the full amount to raise Strength to 3, then it will wait until it does before continuing
//	on to Brawl.
//
//	The thing to keep in mind is that you probably want to keep feeding it values to spend points on, so you
//	should usually put each desired stat in the list multiple times, in different places, until it maxes out.
//
//
//  Format:
//
//	LevelingTemplate
//	{
//		"Name"			""
//		"InternalName"		""
//		
//		"ClanDependency"		"<Character_Template>"
//		"Dependency"		"<Trait> <Equality_Comparison> <Value>"	// A requirement for this template to be valid/process
//		"Dependency"		"Attrib_Order == Physical_Mental_Social"
//
//		LevelGroup			// Any number (0 to N) of groups that get processed in-order
//		{
//			"Dependency"		"<Trait> <Equality_Comparison> <Value>"	// A requirement for this LevelGroup to be processed
//
//			Level { "<Trait>"		"<Value>" }	// Any number of levelling values, that get processed in order
//			Level { "Strength"	"2" }
//		}
//
//		Level { "<Trait>"		"<Value>" }
//
//	}
//
//
// ---------------------
// ---------------------
